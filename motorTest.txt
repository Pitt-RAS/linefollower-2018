
int PWMOutputPin = 1;
int x = 0;

void setup() {
  // put your setup code here, to run once:
  Serial.begin(9600);
  pinMode(PWMOutputPin, OUTPUT);
}

void loop() {
  // put your main code here, to run repeatedly:
  while(x<225){
    analogWrite(PWMOutputPin, 255);
    delay(1000);
    x += 10;
    Serial.println(x);  
  }
  
}
